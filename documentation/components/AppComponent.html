import { Component } from '@angular/core';
import { CommonService } from './services/common.service';
import { StorageService } from './services/storage.service';
import { WebsocketService } from './services/websocket.service';
import {Socket} from 'ngx-socket-io';
import { MasterService } from './modules/master.service';

@Component({
  selector: 'app-root',
  templateUrl: './app.component.html',
  styleUrls: ['./app.component.scss']
})
export class AppComponent {
  title = 'lmsapp';
  receivedAllDevice:any;

  constructor (
    private commonService: CommonService,
    private masterService:MasterService,
    private store: StorageService,
    private ws: WebsocketService,
    private socket: Socket
    ) {
      if(this.store.get('userData')?.user?._id){
        let loginForm : any = {};
        loginForm.user_id = this.store.get('userData')?.client_config?.gpsId;
        loginForm.password =  this.store.get('userData')?.client_config?.gpsPwd;
        loginForm.request = 'authentication';
        let req : any = JSON.stringify(loginForm);
        this.socket.fromEvent('message').subscribe((msg: any)=>{
          if(msg.request != 'authentication') return;
              let res = JSON.parse(msg);
              console.log(res);
        });
        this.socket.emit('message',req);
        this.socketResponse();
        this.getDevices();
      }
    }

    getDevices(){
      this.masterService.getGpsReports({},'devices').subscribe((res: any) => {
          this.receivedAllDevice = res.data;
        });
    }

    stopFeed(sDevice: any,device:any) {
      if(!device) return;
      let req : any = {};
      req.device_id = sDevice.imei;
      req.device_type = sDevice.device_type;
      req.user_id = this.store.get('userData').client_config.gpsId;
      req.request= 'stop_feed';
      this.ws.sendMessage(req);
    }
    
    reconnectRequest() {
    const str = window.location.hash;
     if(str && str.startsWith('#/home/tracking/mapView/liveTrack')) {
      const device_imei = str.split('/')[5];
      //let allDevices = this.store.get('alldevices');
      let device = this.receivedAllDevice && this.receivedAllDevice.length && this.receivedAllDevice.find((item: any) => item.imei == device_imei);
      if(!device) return;
      let stopDevice = this.store.get('lastDevice');
      if(stopDevice && device.imei != stopDevice.imei) {
        this.stopFeed(stopDevice,device);
      }
      this.store.put('lastDevice',device);
      let req : any = {};
      req.device_id = device && device.imei;
      req.device_type = device && device.device_type;
      req.user_id = this.store.get('userData').client_config.gpsId;
      req.request = 'live_feedV2';
      this.ws.sendMessage(req);
     }
    }

    socketResponse() {
      const mContext = this;
      this.socket.on('connect',function (args:any) {
        console.info("successful connection",args);
      });
      this.socket.on("connect_error",function(args:any){
    	  console.info("connect_error",args);
      });
      this.socket.on("connect_timeout",function(args:any){
    	  console.info("connect_timeout",args);
      });
      this.socket.on("reconnect",function(args:any){
        mContext.reconnectRequest();
        console.info("reconnect",args);
      });
      this.socket.on("reconnect_attempt",function(args:any){
        mContext.reconnectRequest();
    	  console.info("reconnect_attempt",args);
      });
      this.socket.on("reconnecting",function(args:any){
        console.info("reconnecting",args);
      });
      this.socket.on("reconnect_error",function(args:any){
    	  console.info("reconnect_error",args);
      });
      this.socket.on("reconnect_failed",function(args:any){
    	  console.info("reconnect_failed",args);
      });
      this.socket.on("disconnect",function(args:any){
        console.info("disconnect",args);
      });
    }
  loadingObservable = this.commonService.loadingObservable;
}
