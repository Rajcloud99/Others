import {Component, OnInit, ChangeDetectionStrategy, HostListener} from '@angular/core';
import { Router, ActivatedRoute } from '@angular/router';
import {FormBuilder,ReactiveFormsModule,FormGroup,Validators} from '@angular/forms';
import { LoginService } from '../services/login.service';
import {StorageService} from '../services/storage.service';
import { Socket } from 'ngx-socket-io';
declare let window: any;

@Component({
  selector: 'app-login',
  templateUrl: './login.component.html',
  styleUrls: ['./login.component.scss'],
  changeDetection: ChangeDetectionStrategy.OnPush
})
export class LoginComponent implements OnInit {
  hide = true;
  User: any;
  @HostListener("window: resize", ["$event"])
  onResize(event: any) {
    this.innerWidth = window.innerWidth;
  }

  loginForm = this.fb.group({
    userId: [null,[Validators.required]],
    password: [null,[Validators.required]]
  });
  configs: any;
  innerWidth:any;
  loginPath:any;
  constructor(private fb: FormBuilder,
    private route: ActivatedRoute,
    private service: LoginService,
    private store: StorageService,
    private router: Router,
    private socket: Socket) { }

  ngOnInit(): void {
    this.innerWidth = window.innerWidth;
    this.loginPath = "home/tracking/mapListView";
    this.configs = this.store.get('userData')?.configs;
    if(this.store.get('loggedIn')) {
      this.router.navigate(['./home']);
    }
    if(this?.configs?.loginPath || (this?.configs && this.loginPath)) {
      this.router.navigateByUrl(this?.configs?.loginPath || this.loginPath);
    }
  }

  submitForm():void {
    let formData = this.loginForm.getRawValue();
    if(this.store.get('loggedIn')) {
      formData.userId = this.store.get('user');
      formData.password = this.store.get('pwd');
      this.router.navigate(['./home']);
    }
    if(this?.configs?.loginPath || (this?.configs && this.loginPath)) {
      this.router.navigateByUrl(this?.configs?.loginPath || this.loginPath);
    }
    let randomNo = Date.now()+''+Math.round(Math.random()*100);
		formData.request_id = Date.now()+''+Math.round(Math.random()*100);
		formData.validate = 'all';
    this.service.login(formData)
    .subscribe((isOK:any) =>  {
      console.log(isOK);
      if(isOK) {
        this.handleOneSignalUserId(isOK);
        //this.getAllDevices();
        this.router.navigate(['./home']);
        this.User = this.store.get('userData').user;
        // in case of broker memo redirect to create trip page
        if (this.User?.user_type?.indexOf('Broker')+1) {
          this.loginPath = 'home/operation/createTrip';
        }
        if(isOK?.configs?.loginPath || (isOK?.configs && this.loginPath)) {
          this.router.navigateByUrl(isOK?.configs?.loginPath || this.loginPath);
        }
      }
    })
  }

  handleOneSignalUserId(loginResponse:any){
    let that = this;
    if(window && window.OneSignal && window.OneSignal.default){
      window.OneSignal.default.getDeviceState(function(onseSig:any){
        if(onseSig && onseSig.userId){
          console.log('onseSig user data',onseSig);
          let addNotifId:any,addGpsId:any,removeId :any = [], removeGpsId :any = [];
          that.service.getUsertrim({notifId: onseSig.userId,_id:loginResponse.user._id}).subscribe((res: any) => {
              if(res && res.data && res.data.length) {
                res.data.forEach((obj: any)=>{
                  if(obj._id != loginResponse.user._id){
                    removeId.push(obj._id)
                    if(obj.gpsId != loginResponse.client_config.gpsId){
                      console.log('removeId ',obj.gpsId);
                      removeGpsId.push(obj.gpsId);
                    }
                  }else if(obj._id === loginResponse.user._id){
                    if(!(loginResponse && loginResponse.user && loginResponse.user.notifId)) {
                      addNotifId = loginResponse.user._id;
                      addGpsId = loginResponse.client_config.gpsId;
                    }
                  }
                })
              }

              if(!(loginResponse && loginResponse.user && loginResponse.user.notifId)) {
                addNotifId = loginResponse.user._id;
                addGpsId = loginResponse.client_config.gpsId;
              }else if((loginResponse && loginResponse.user && loginResponse.user.notifId) != onseSig.userId){
                // removeId.push(loginResponse.user.notifId);
                addNotifId = loginResponse.user._id;
                console.log('addNotifId',loginResponse.user.notifId);
              }

              if(addNotifId){
                that.service.updateUserNotifid({_id: addNotifId, notifId: onseSig.userId }).subscribe((res: any) => {
                  if(res && res.status) {
                    console.log('updateUserNotifid',res.status)
                  }else{
                    console.log('updateUserNotifid addNotifId',res)
                  }
                });
              }

              if(removeId && removeId.length){
                that.service.removeUserNotifid({_id: removeId }).subscribe((res: any) => {
                  if(res && res.status) {
                    console.log('removeUserNotifid',res.status)
                  }else{
                    console.log('removeUserNotifid',res)
                  }
                });
              }

              if(addGpsId) {
                  console.log('setExternalUserId',addGpsId);
                  window.OneSignal.default.setExternalUserId(addGpsId);
              }

              if(removeGpsId && removeGpsId.length){
                  removeGpsId.forEach((exId:any)=> {
                    console.log('removeExternalUserId',exId);
                    window.OneSignal.default.removeExternalUserId(exId);
                  })
              }

            });

        }
      });
    }
  }

  getAllDevices(): void {
    let req: any = {};
    req.request = "gpsgaadi_by_uid";
    req.version = 2;
    req.type_url = 'trucku';
    req.selected_uid = this.store.get('userData').client_config.gpsId;
    req.login_uid = this.store.get('userData').client_config.gpsId;
    this.service.getDevices(req).subscribe((res: any)=>{
      this.store.put('alldevices',res);
    });
  }
}
