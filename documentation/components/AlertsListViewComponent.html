  search (download : any) {
    this.clean(this.filterObj);
    let req : any = {row_count : 15,no_of_docs : 15,};
    if (this.filterObj.from_date) {
      req.from = this.setHours(this.filterObj.from_date, 0, 0, 0).toISOString();
    }
    if (this.filterObj.to_date) {
      req.to = this.setHours(this.filterObj.to_date, 23, 59, 59).toISOString();
    }
    if(this.filterObj.code && this.filterObj.code.length && this.filterObj.code.length>0){
      req.code = this.filterObj.code;
    }

    if (this.filterObj.vehileobj) {
      if (this.filterObj.vehileobj.device_imei) {
        req.imei = [this.filterObj.vehileobj.device_imei];
      }
      if (this.filterObj.vehileobj.vehicle_reg_no){
        req.regVeh = [];
        req.regVeh.push({
          reg_no: this.filterObj.vehileobj.vehicle_reg_no,
          device_imei: this.filterObj.vehileobj.device_imei
        });
        if( !req.regVeh[0].device_imei){
          delete req.regVeh[0].device_imei;
        }
      }
    }
    if (download) {
      req.download = true;
      req.no_of_docs = 2000
    }
    // this.clean(this.filterObj);
    this.masterService.getAllAlerts({row_count: 8, skip:1, ...req}).subscribe((res: any) => {
      if(res && !download){
        this.receivedAlertsData = res;
        this.scrollTable = document.getElementsByClassName("ant-table-body");
        this.scrollTable[0]?.addEventListener('scroll',() => {
        this.getDataOnScroll();
      });
      }else if(download){
        if (res.url) {
          this.commonService.downloadFileWithUrl(res.url);
        } else {
          // swal("warning", res.data.message, "warning");
        }
      }

    });
  }

   setHours(date: any, hour: any, minutes: any, second: any){
    return new Date(new Date(date).setHours(hour, minutes, second))
  }

  downloadReport() {
    this.search(true);
  }

  onSearchVeh(value: any) {
    this.receivedAllVehicles = [];
    if(value && value.length> 1) {
      this.masterService.getAllVehicles({vehicle_no: value}).subscribe((res: any) => {
        this.receivedAllVehicles = res;
      });
    }
  }

  getDataOnScroll(){
    if (true) {
      let divElement: any = document.getElementsByClassName('ant-table-body')[0] || document.getElementById('myDIV');
      // console.log('divElement', divElement);
      if ((divElement.scrollTop + divElement.clientHeight) >= divElement.scrollHeight) {
        if (this.stopScrolling === false) {
          if (this.timer) {
            clearTimeout(this.timer);
          }
          const mContext = this;
          this.timer = window.setTimeout( ()=> {
            mContext.masterService.getAllAlerts({ no_of_docs: 15,row_count:8, skip: mContext.pageNumber,
              ...this.filterObj }).subscribe(data => {
              if (data) {
                if (data.length=== 0) {
                  mContext.stopScrolling = true;
                } else {
                  mContext.receivedAlertsData = [...mContext.receivedAlertsData, ...data];
                  mContext.pageNumber = mContext.pageNumber + 1;
                }
              }
              return;
            });
          }, 400);
        }
      }
    }
  }

  selectAlerts(item: any) {
    this.receivedAlerts = item;
    this.selectedAlertsRowId = item._id;
  }

  reset(){
    this.filterObj.from_date = null;
    this.filterObj.to_date = null;
    this.filterObj.vehileobj = null;
    this.filterObj.code = null
    this.isMobFilterVisible = false;
    this.search(false);
  }
  clean(obj : any) {
    for (var propName in obj) {
      if (obj[propName] === null || obj[propName] === undefined) {
        delete obj[propName];
      }
    }
    // return obj;
  }


  addTrip(){}
