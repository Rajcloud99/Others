import {Component, ElementRef, HostListener, OnInit, ViewChild} from '@angular/core';
import {FormArray, FormBuilder, Validators} from "@angular/forms";
import {CommonService} from "../../../services/common.service";
import {MasterService} from "../../master.service";
import {ActivatedRoute, Router} from "@angular/router";
import {ConstantService} from "../../../services/constant.service";
import {NzModalService} from "ng-zorro-antd/modal";

@Component({
  selector: 'app-milestone',
  templateUrl: './milestone.component.html',
  styleUrls: ['./milestone.component.scss']
})
export class MilestoneComponent implements OnInit {
  innerWidth: any;// stores width of any screeen
  routedata :any;
  aMilestone:any={};
  mode:boolean = false;
  milestonepopupvisible :boolean = false;
  selectedItem: any;
  receivedLocation:any;
  index: any;
  Editmode:boolean = false;
  aCategory:any = [
    {
      name: "city",
      value: "city",
    },
    {
      name:"toll",
      value:"toll"
    },
    {
      name:"parking",
      value:"parking"
    },
    {
      name:"hotel",
      value:"hotel"
    }
  ];
  constructor(
    private fb: FormBuilder,
    private commonService: CommonService,
    private masterService: MasterService,
    private router: Router,
    private route: ActivatedRoute,
    private constantService: ConstantService,
    private modal: NzModalService,

  ) {
    this.route.params.subscribe((queryParam) => {
      if (queryParam) {
        const currentState = this.router.getCurrentNavigation();
        this.routedata = currentState && currentState.extras.state;
        if(this.routedata && this.routedata.route__id){
          this.mode = true;
        }
      }
    });
  }
  @HostListener("window: resize", ["$event"])
  onResize(event: any) {
    this.innerWidth = window.innerWidth;
  }

  ngOnInit(): void {
    this.commonService.setNavTittleAndMenu('Milestone');
    this.innerWidth = window.innerWidth;
    if(this.routedata?.milestone &&  this.routedata?.milestone.length) {
      this.receivedLocation= [];
      for (let data of this.routedata?.milestone) {
        this.receivedLocation.push({name:data.name,location:{latitude:data.location.lat,longitude:data.location.lng},geozone:data.geozone});
        this.addmilestoneRows(data);
      }
    }
  }
  get f() {
    return this.MilestoneForm.controls;
  }
  get milestones() {
    return this.MilestoneForm.controls['milestone'] as FormArray;
  }
  MilestoneForm = this.fb.group({
    milestone: <FormArray>this.fb.array([]),
  });
  addmilestoneRows(data: any) {
    if(this.MilestoneForm.value.milestone && this.MilestoneForm.value.milestone.length > 19){
     return  this.commonService.error("Only 20 milestones allowed");
    }
    (<FormArray>this.milestones).push(
      this.fb.group({
        category: [(data && data.category)],
        name: [(data && data.name)],
        geozone:[(data && data.geozone) || []],
        tat_hr: [(data && data.tat_hr)],
        tat_min: [(data && data.tat_min)],
        location: this.fb.group({
          lat: [(data && data.location && data.location.lat)],
          lng: [(data && data.location && data.location.lng)],
        }),
        halt_d: [(data && data.halt_d)],
        tat_km: [(data && data.tat_km)],
      })
    );
  }
  editmilestonedata(index:number,data: any) {
    (<FormArray>this.milestones).setControl(index,
      this.fb.group({
        category: [(data && data.category)],
        geozone:[(data && data.geozone) || []],
        name: [(data && data.name)],
        tat_hr: [(data && data.tat_hr)],
        tat_min: [(data && data.tat_min)],
        location: this.fb.group({
          lat: [(data && data.location && data.location.lat)],
          lng: [(data && data.location && data.location.lng)],
        }),
        halt_d: [(data && data.halt_d)],
        tat_km: [(data && data.tat_km)],
      })
    );
  }

  deletemilestoneRow(index: number) {
    if (this.milestones.controls.length) {
      (<FormArray>this.f.milestone).removeAt(index);
      return true;
    } else return false;
  }
  clean(obj: any) {
    for (let propName in obj) {
      if (
        obj[propName] === null ||
        obj[propName] === undefined ||
        obj[propName] === ''
      ) {
        delete obj[propName];
      }
    }
  }

  onSubmit(){
    if(this.MilestoneForm.invalid){
      this.commonService.error("all mandatory Fields  is required");
      return;
    }
    let formdata :any = {};
    formdata.milestone = this.milestones.value;
    if(formdata.milestone.length){
        this.cleaMilestone(formdata.milestone);
    }
      formdata.route__id = this.routedata.route__id  || this.routedata._id;
      formdata.route_distance = this.routedata.route_distance;
      formdata.route_name = this.routedata.route_name || this.routedata.name;
      formdata.category = 'tracking';
    if(this.mode){
      this.updatemilestone(formdata);
    }else{
      this.addmilestone(formdata);
    }

  }
  addmilestone(request:any){
      if (request) {
        this.masterService.addmilestone(request).subscribe((data: any) => {
          if (data) {
            this.router.navigateByUrl('home/master/transportRoute');
          }
        });
      }
  }
  updatemilestone(request:any){
    if (request) {
      this.masterService.updatemilestone(this.routedata._id,request).subscribe((data: any) => {
        if (data) {
          this.router.navigateByUrl('home/master/transportRoute');
        }
      });
    }
  }
  cleaMilestone(obj: any) {
    obj.forEach((item: any) => {
      for (let propName in item) {
        if (
          item[propName] === null ||
          item[propName] === undefined ||
          item[propName] === ''
        ) {
          delete item[propName];
        }
      }
    });
  }
  editmilestonepopup( index:number,data:any): any {
    this.Editmode = true;
    this.aMilestone = {};
    if (data) {
      this.index = index;
      this.receivedLocation = [{name:data.name,location:{latitude:data.location.lat,longitude:data.location.lng},geozone:data.geozone || []}];
      this.aMilestone.name = this.receivedLocation[0];
      this.aMilestone.category = data.category;
      this.aMilestone.geozone = this.aMilestone.geozone;
      this.aMilestone.tat_hr = data.tat_hr;
      this.aMilestone.tat_min = data.tat_min;
      this.aMilestone.tat_km = data.tat_km;
      this.aMilestone.halt_d = data.halt_d;
      this.aMilestone.lat = data.location.lat;
      this.aMilestone.lng = data.location.lng;
      this.milestonepopupvisible = true;
    }
  }
  addmilestonepopup(): any {
    if(this.MilestoneForm.value.milestone && this.MilestoneForm.value.milestone.length > 19){
        return   this.commonService.error("Only 20 milestones allowed");
    }
    this.Editmode = false;
    this.receivedLocation =[];
    this.aMilestone = {};
      this.milestonepopupvisible = true;
  }


  handleCancel(){
    this.aMilestone={};
    this.milestonepopupvisible = false;
  }
  handleOk() {
    if (this.aMilestone) {
      let aMilestone: any = {};
      if(!this.aMilestone.category){
       return  this.commonService.error("please Choose Category");
      }
      if(!this.aMilestone.name){
        return  this.commonService.error("milestone name is mandatory ");
      }
      if(!(this.aMilestone.lat )){
        return  this.commonService.error("latitude is mandatory ");
      }
      if(!(this.aMilestone.lng )){
        return  this.commonService.error("longitude is mandatory ");
      }
      aMilestone.category = this.aMilestone.category;
      aMilestone.name = this.aMilestone.name.name;
      aMilestone.geozone = this.aMilestone.name.geozone || [];
      aMilestone.tat_hr = this.aMilestone.tat_hr;
      aMilestone.tat_min = this.aMilestone.tat_min;
      aMilestone.tat_km = this.aMilestone.tat_km;
      aMilestone.halt_d = this.aMilestone.halt_d;
      aMilestone.location = {
        lat: this.aMilestone.lat,
        lng:this.aMilestone.lng
      }
      if (this.Editmode) {
        this.editmilestonedata(this.index, aMilestone)
      } else {
        this.addmilestoneRows(aMilestone);
      }
      this.milestonepopupvisible = false;
    }
    this.aMilestone={};
  }

  getLandmark (query:any) {
    if(query.length<2 ){
      return;
    }
    let reqParam={
      type_url: 'gps_url',
      sort: {_id: -1},
      name:query,
      // type:'landmark',
    }
    this.clean(reqParam);
    this.masterService.getAllLandmark(reqParam).subscribe((res: any) => {
      if(res) {
        this.receivedLocation= res.data;
      }
    });
  }
    setlocation ($event:any) {
    this.receivedLocation = [];
    if($event.geozone && $event.geozone[0]){
      this.aMilestone.lat =$event.geozone[0].latitude;
      this.aMilestone.lng = $event.geozone[0].longitude;
    }else {
      this.aMilestone.lat = $event.location.latitude;
      this.aMilestone.lng = $event.location.longitude;
    }
    }
  setlocationpoint($event:any,index:number){
    const data = this.receivedLocation.find((y:any) => y.name == $event);
    let formdata:any={};
    formdata.name = data.name;
    formdata.category =  this.MilestoneForm.value.milestone[index].category;
    if(data.geozone && data.geozone[0]){
      formdata.location = {lat :data.geozone[0].latitude,lng: data.geozone[0].longitude};
      formdata.geozone = data.geozone;
    }else{
      formdata.location = {lat :data.location.latitude,lng: data.location.longitude};
      formdata.geozone = [];
    }

    formdata.tat_hr = this.MilestoneForm.value.milestone[index].tat_hr;
    formdata.tat_min = this.MilestoneForm.value.milestone[index].tat_min;
    formdata.tat_km = this.MilestoneForm.value.milestone[index].tat_km;
    formdata.halt_d = this.MilestoneForm.value.milestone[index].halt_d;
    this.editmilestonedata(index,formdata);

  }
}
