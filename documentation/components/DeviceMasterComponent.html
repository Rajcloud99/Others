import { Component, ElementRef, HostListener, OnInit, ViewChild } from '@angular/core';
import { MasterService } from '../master.service';
import { Router } from '@angular/router';
import { CommonService } from 'src/app/services/common.service';
import { MoreMenu } from 'src/app/shared/modal/moreMenu.modal';
import { ObjToCsvService } from 'src/app/services/obj-to-csv.service';
import { DatePipe } from '@angular/common';


@Component({
  selector: 'app-device-master',
  templateUrl: './device-master.component.html',
  styleUrls: ['./device-master.component.scss']
})
export class DeviceMasterComponent implements OnInit {
  innerWidth: any;// stores width of any screeen
  receivedData: any;
  receivedAllDevice: any;
  filterIMEI:any;
  selectStatus=['online','offline','inactive'];
  stopScrolling = false;
  pageNumber = 2;
  apiType='devices';
  timer: any;
  filterObj = {
    imei: '',
    status:''
  };
  selectedRowId: any;
  selectedItem: any;
  isMobFilterVisible = false;

  scrollTable!: HTMLCollectionOf<Element>;
  @HostListener("window: resize", ["$event"])
  onResize(event: any) {
    this.innerWidth = window.innerWidth;
  }
  constructor(
    private masterService: MasterService,
    private router: Router,
    private objToCsvService: ObjToCsvService,
    private datePipe: DatePipe,
    private commonServ : CommonService
  ) { }

  ngOnInit(): void {
    this.innerWidth = window.innerWidth;
    let _moreMenu: MoreMenu[] = [
      {
        tittle: "Filter",
        iconName: "fa-filter",
        isVisible: true,
        clickEventMethode:()=> {
          this.showFilter();
        },
      },

      {
        tittle: "Download CSV",
        iconName: "fa-file",
        isVisible: true,
        clickEventMethode:()=> {
          this.downloadCsv(this.receivedData);
        },
      },
      
    ];

    this.commonServ.setNavTittleAndMenu("Registered Device",_moreMenu);
    this.search();
  }

  selectItem(item: any) {
    this.selectedItem = item;
    this.selectedRowId = item._id;
    this.commonServ.hideShowMenu(["Filter","Download CSV"],true);
  }

  showFilter() {
    this.isMobFilterVisible = true;
  }


  reset () {
    this.filterObj.imei = '';
    this.filterObj.status='';
    this.receivedData=this.receivedAllDevice;
  }

  onSearch(value?: any) {
    let imei=this.filterObj.imei;
    let status=this.filterObj.status;
    if(imei || status) {
      if(imei){
        this.receivedData=this.receivedAllDevice.filter((d:any)=>{
          return d.imei.includes(imei);
        })
      }else if(status) {
        this.receivedData=this.receivedAllDevice.filter((d:any)=>{
          return d.status.includes(status);
        })
      }
      
    }else{
      this.receivedData=this.receivedAllDevice;
    }
  }

  downloadCsv(aData: any) {
    let cnt = 1;
		this.objToCsvService.convertToCsv('Registered Devices', [
			'S.NO.',
			'DEVICE IMEI',
			'CURRENT LOCATION',
			'LAST ONLINE DATE',
			'STATUS',
		], aData.map( (d:any) => {
			let arr = [];

			try{
				arr.push(cnt++ || 0 );

			}catch(e){
				arr.push(0);
			}

			try{
				arr.push(d?.imei || 'NA');
			}catch(e){
				arr.push("NA");
			}

			try{
				arr.push(d?.addr.replace(/,/g,' ') || 'NA');
			}catch(e){
				arr.push("NA");
			}

      try{
				arr.push(this.datePipe.transform(d?.datetime, 'dd-MMM-yyyy \'at\' h:mma') || 'NA');
			}catch(e){
				arr.push("NA");
			}
			
			try{
				arr.push(d?.status.replace(/,/g,' ') || 'NA');
			}catch(e){
				arr.push("NA");
			}

			return arr;

		}));
  }


  search () {
    this.masterService.getGpsReports({imei:this.filterObj.imei, no_of_docs: 15},this.apiType).subscribe((res: any) => {
      if(res) {
        this.receivedAllDevice=res.data;
        this.receivedData = this.receivedAllDevice;
        //this.scrollTable = document.getElementsByClassName("ant-table-body");
        // this.scrollTable[0]?.addEventListener('scroll',() => {
        // this.getDataOnScroll();
        // })
      }

    });
}

getDataOnScroll() {
  if (false) {
    let divElement: any = document.getElementsByClassName('ant-table-body')[0] || document.getElementById('myDIV');
    if ((divElement.scrollTop + divElement.clientHeight) >= divElement.scrollHeight) {
      if (this.stopScrolling === false) {
        if (this.timer) {
          clearTimeout(this.timer);
        }
        const mContext = this;
        this.timer = window.setTimeout( ()=> {
          mContext.masterService.getGpsReports({ no_of_docs: 10, skip: mContext.pageNumber, imei: this.filterObj.imei },this.apiType).subscribe(data => {
            if (data.data) {
              if (data.data.length=== 0) {
                mContext.stopScrolling = true;
              } else {
                mContext.receivedData = [...mContext.receivedData, ...data.data];
                mContext.pageNumber = mContext.pageNumber + 1;
              }
            }
            return;
          });
        }, 400);
      }
    }
  }
}

}
