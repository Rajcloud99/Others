import {Component, OnInit, ViewContainerRef,HostListener, ViewChild, ElementRef} from '@angular/core';
import {CommonService} from "../../../services/common.service";
import {FormBuilder, Validators, FormArray, FormControl, FormGroup} from "@angular/forms";
import {MasterService} from "../../master.service";
import {ActivatedRoute, Router} from "@angular/router";
import {StorageService} from "../../../services/storage.service";
// import {NzModalService} from "ng-zorro-antd/modal";
import * as XLSX from 'xlsx';

import { MoreMenu } from 'src/app/shared/modal/moreMenu.modal';

@Component({
  selector: 'app-sensor-upsert',
  templateUrl: './sensor-upsert.component.html',
  styleUrls: ['./sensor-upsert.component.scss']
})
export class SensorUpsertComponent implements OnInit {

  categoryList: any = ['Fuel', 'Temperature', 'Pressure'];
  mode: any;
  getId: any;
  innerWidth: any; // stores width of any screeen
  sensorForm : any;
  selectedItem: any;
  editIndex:any;
  receivedData: any;
  calibrationData:any=[];
  selectedRowId:any;
  level:any;
  value:any;
  showHideCalib=true;
  editMode:boolean=false;

  @ViewChild("fileInput", { static: true })
  fileInput!: ElementRef;

  @HostListener('window: resize', ['$event'])
  onResize(event: any) {
    this.innerWidth = window.innerWidth;
  }

  constructor(
    private fb: FormBuilder,
    private commonService: CommonService,
    private masterService: MasterService,
    private router: Router,
    private route: ActivatedRoute,
    private storageService: StorageService,
    // private modal: NzModalService,
    private viewContainerRef: ViewContainerRef,
  ) {
    this.sensorForm = this.fb.group({
      s_id: [null,[Validators.required]],
      company: [null,[Validators.required]],
      device: [null,[Validators.required]],
      out_unit: [null,[Validators.required]],
      conversion_fact: [null,[Validators.required]],
      ver: [null,[Validators.required]],
      fill_diff: [null,[Validators.required]],
      drain_diff: [null,[Validators.required]],
    });

    this.route.params.subscribe(queryParam => {
      if(queryParam._id) {
        this.mode = 2;
        this.getId = queryParam._id;
        const currentState = this.router.getCurrentNavigation();
        this.receivedData = currentState && currentState.extras.state;
        this.fillFormdata(this.receivedData);
        this.calibrationData=this.receivedData?.calib||[];
      } else {
        this.mode = 1;
      }
    })

  }

  ngOnInit(): void {
    this.innerWidth = window.innerWidth;
    let _moreMenu:MoreMenu[]=[
      {
        tittle:"Upload",
        iconName:"fa-upload",
        isVisible:true,
        clickEventMethode:()=>{
          this.fileInput.nativeElement.click();
        }
      },

      {
        tittle:"Download Sample",
        iconName:"fa-download",
        isVisible:true,
        clickEventMethode:()=>{
          this.downloadSample();
        }
      }

    ];
    this.commonService.setNavTittleAndMenu("Sensor-Upsert",_moreMenu);
  }

  fillFormdata (data: any) {
    this.sensorForm.patchValue({
      s_id:  data && data.s_id,
      company:data && data.company,
      device: data && data.device,
      out_unit: data && data.out_unit,
      conversion_fact: data && data.conversion_fact,
      ver: data && data.ver,
      fill_diff: data && data.fill_diff,
      drain_diff: data && data.drain_diff
    })
  }

  selectItem(item: any) {
    this.selectedItem = item;
    this.selectedRowId = item._id;
  }

  onSubmit(){
    if (this.sensorForm.invalid || !this.calibrationData.length) {
      this.commonService.error("Please fill mandatory fields");
      return;
    }
    let values = this.sensorForm.value;

    values={...values,"category":"Fuel","calib":this.calibrationData}

    console.log(values);

    if (this.mode === 2 && this.getId) {
      this.updateData(values);
    } else {
      this.sensorForm.value.category = 'Fuel';
      this.saveData(values);
    }
  }

  saveData(formData: any) {
    formData.type_url = 'gps_url';
    formData.user_id = this.storageService.get('userData').client_config.gpsId;
    this.masterService.addSensor(formData).subscribe((data: any) => {
      console.log(data);
      if (data) {
        this.router.navigateByUrl('home/tracking/sensor');
      }
    });
  }


  updateData(formData: any) {
    this.masterService.editSensor(this.getId,formData).subscribe((data: any) => {
      if(data) {
        this.router.navigateByUrl('home/tracking/sensor');
      }
    });
  }

  downloadSample(){
    const readyToExport = [
      {lvl: '', val: ''}
    ];
    this.commonService.exportAsExcelFile(readyToExport,'Calibration');
  }


  editCalib(index:any){
    if (this.calibrationData.length) {
    this.level=this.calibrationData[index].lvl;
    this.value=this.calibrationData[index].val;
    this.editIndex=index;
    }
  }

  deleteCalib(index:any){
    if (this.calibrationData.length) {
      this.calibrationData.splice(index, 1);
    }
  }

  validateDocBeforeUpload(event:any){
    let file=event.target.files[0];
    const fileExtension=file.name.replace(/^.*\./, '');
    if(!(fileExtension === 'xls' || fileExtension === 'xlsx')){
      this.commonService.error('Only Excel file allowed');
      return;
    }
    const reader = new FileReader();
    reader.readAsBinaryString(file);
    reader.onload = (event:any) => {
      let binaryData = event.target.result;
      let workBook = XLSX.read(binaryData, { type: 'binary' });
      workBook.SheetNames.forEach((sheet:any)=>{
        let data : any =[];
        data = XLSX.utils.sheet_to_json(workBook.Sheets[sheet]);
        for (let i = 0; i < Math.min(21,data.length); i++) {
          this.addCalibration(data[i]?.lvl,data[i]?.val);
        }
      });
    }
  }

  addCalibration(level:any,value:any){
    if(!level || !value){
      this.commonService.error('Level and Value both are mandatory!');
      return;
    }else if(this.editMode) {
      let editId=this.calibrationData[this.editIndex]?._id;
      this.calibrationData[this.editIndex]={'lvl':level,'val':value,'_id':editId};
      this.level='';
      this.value='';
      this.editMode=false;
    }else if(this.calibrationData.length>19){
      this.commonService.error('Only 20 records allowed.');
      return;
    }else {
      let obj={'lvl':level,'val':value};
      this.level='';
      this.value='';
      this.calibrationData.push(obj);
    }

  }

}
