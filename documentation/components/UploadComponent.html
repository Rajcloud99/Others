    for (let i = 0; i < this.accordions[5].child.length; i++) {
      let componentType = this.accordions[5].child[i];
      let imgName = this.accordions[5].child[i].imgName;

  checkBeforeProcess(content: any){
    let UnLoadList=['f1u','f2u','f3u', 'f4u', 'f5u','b1u','b2u','b3u','b4u','b5u','l1u','l2u','l3u','l4u','l5u', 'l6u','l7u','l8u','l9u',
      'l10u','r1u','r2u','r3u','r4u','r5u','r6u','r7u','r8u','r9u','r10u'];
      let index = this.accordions[5].child.findIndex((o:any) => o.imgName === content?.imgName);
      let loadContent=this.accordions[5].child[index-1];
      if(UnLoadList.includes(content?.imgName) && content?.scanVal !== loadContent?.scanVal){
        const modal: NzModalRef = this.modal.create({
          nzTitle: 'Warning',
          nzContent: `Barcode Value Not Matched with ${loadContent?.name}`,
          nzFooter:[
            {
             label: 'Close',
             type: 'primary',
             danger: true,
             onClick: () => {modal.destroy();
              content.enableSave = false;
              content.manualBarcode = true;
            }
            },
            {
              label: 'Proceed',
              type: 'primary',
              onClick: ()=>this.processFile(content.url, content,modal)
            },
            {
              label: 'Retake',
              type: 'primary',
              onClick: ()=>{this.scanBarCode(content,modal);
                content.enableSave = false;
                content.manualBarcode = true;
              }
            }
          ],
          nzMaskClosable:false,
          nzClosable:false,
        });
      }else{
        this.processFile(content.url, content);
      }
  }

  isUniqueTag(content:any) : boolean{
    let val=content?.scanVal;
    let namef = content?.imgName;
    let uniqueList=[];
    let LoadList=['f1l','f2l', 'f3l', 'f4l', 'f5l','b1l','b2l','b3l','b4l','b5l', 'l1l','l2l','l3l','l4l','l5l', 'l6l','l7l','l8l','l9l',
      'l10l','r1l','r2l','r3l','r4l','r5l','r6l','r7l','r8l','r9l','r10l'];
    if(LoadList.includes(namef)){
      for (let i = 0; i < this.accordions[5].child.length; i+=2) {
        if(val == this.accordions[5].child[i]?.scanVal && namef!=this.accordions[5].child[i]?.imgName){
          this.common.error(`Barcode tag already exist in ${this.accordions[5].child[i]?.name}`);
          content.enableSave = false;
          content.manualBarcode = true;
          return false;
        }else {
          uniqueList.push(content);
        }
      }
      if(uniqueList.length){
        return true;
      }
    }
    return true;
  }

  updateBarCodeCount(id:any) {
    if(!id){
      return;
    }
    let match=[];
    let unMatch=[];
    let NotMatch=[];
    let LoadList=['f1l','f2l', 'f3l', 'f4l', 'f5l','b1l','b2l','b3l','b4l','b5l', 'l1l','l2l','l3l','l4l','l5l', 'l6l','l7l','l8l','l9l',
      'l10l','r1l','r2l','r3l','r4l','r5l','r6l','r7l','r8l','r9l','r10l'];
    let UnLoadList=['f1u','f2u','f3u', 'f4u', 'f5u','b1u','b2u','b3u','b4u','b5u','l1u','l2u','l3u','l4u','l5u', 'l6u','l7u','l8u','l9u',
      'l10u','r1u','r2u','r3u','r4u','r5u','r6u','r7u','r8u','r9u','r10u'];
    for (let i = 0; i < this.accordions[5].child.length; i++) {
      let tag = this.accordions[5].child[i];
      if(tag?.url!=""){
        if(tag?.status==='Match'){
          match.push(i);
        }
        if(LoadList.includes(tag.imgName) && this.accordions[5].child[i+1]?.status!='Match'){
            unMatch.push(i);
        }
        if(UnLoadList.includes(tag.imgName) && tag?.status ==='Not Match'){
          NotMatch.push(tag?.name);
        }
      }
    }

    if(NotMatch.length){
      let msg = NotMatch.join(", ");
      this.common.warning('Tag Not Matched',`${msg}`);
    }
    if((match.length || unMatch.length) && match.length != this.aParentCompData.match && unMatch.length != this.aParentCompData.unMatch){
      this.updateTagCount(id,match.length,unMatch.length);
    }
  }

  processFile(filePath: any, content: any, modal?:NzModalRef) {
    if(modal){
      modal.destroy();
    }
    content.enableSave = false;
    content.manualBarcode = false;
    if(!content.localFile && content.scanVal){
      //No file chnage from camera
      this.imageService.uploadTag(content).subscribe(
        (res) => {
          this.onSuccess();
        },
        (err) => {
          this.onError();
        })

    }else{
      this.imageService.readImage(filePath, (error: any, base64Img: any) => {
        this.selectedFile.pending = true;
        this.imageService.uploadImage(base64Img, content).subscribe(
          (res) => {
            this.onSuccess();
          },
          (err) => {
            this.onError();
          })
      });
    }
  }
