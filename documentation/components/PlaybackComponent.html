  constructor(
    private route: ActivatedRoute,
    private router: Router,
    private masterService: MasterService,
    private storageService: StorageService,

    private objToCsvService: ObjToCsvService,

  ) {
    this.route.params.subscribe((queryParam) => {
      if (queryParam) {
        const currentState = this.router.getCurrentNavigation();
        this.selectedTripData = currentState && currentState.extras.state;
      }
    });
    this.truck_path = this.constantService.constants.truck_path;
    // this.idle_path = this.constantService.constants.idle_path;

  ngOnDestroy(): void {
    if(this.timerRef) {
      clearTimeout(this.timerRef);
      this.timerRef = undefined;
    }
  }

  ngOnInit(): void {
    this.getDevices();
    if(this.selectedTripData?.status == 'OK' && this.selectedTripData?.data?.length) {
      let res = this.selectedTripData;
      this.showTruck = true;
      this.vehData = res;
      this.playBackData = res.data;
      this.vehData.tot_dist = this.vehData?.tot_dist / 1000;
      this.drawFlags();
      this.drawRoute();
      this.preProcessing();
    } else  {
      if(this.selectedTripData?.tripBase == 'Sim Based') {
        this.getSimBasePlaybackData();
      } else  {
        this.getPlayBackData();
      }
    }

    this.truckIcon= {
      path: this.truck_path,
      fillColor:"#15e425",
      fillOpacity: 0.8,
      strokeWeight: 0.5,
      scale: 0.5,
      rotation: 0,

      fillColor:"#0000ff",
      fillOpacity: 1,
      strokeWeight: 1,
      strokeColor: "#FFFFFF",
      scale: 0.055,
    };
  }

    }

      return this.commonService.errorModal('Error','There is no data to see in listView');
    }
    this.router.navigateByUrl('home/operation/playback-list', {state: this.playBackData});
  }

  back() {
    this.router.navigateByUrl(
      'home/operation/tripDetails/view/' + this.selectedTripData._id,
      { state: this.selectedTripData }
    );
  }

  downloadCsv (aData: any) {
    if(this.selectedTripData?.tripBase =='Sim Based') {
      return this.getSimBasePlaybackData(true);
    }
    let cnt = 1;
		this.objToCsvService.convertToCsv('PlaybackSheet',[
			'S.No',
			'Start Time',
			'End Time',
			'Location',
			'Latitude',
			'Longitude',
			'Nearest Landmark',
			'Speed(Kmph)',
			'Duration(Hour)',
			'Distance(Kms)'
		], aData.map( (o: any) => {
			let arr = [];
			try {
				arr.push(cnt++ || 0);
			} catch (e) {
				arr.push(0);
			}

			try {
				arr.push(this.datePipe.transform(o.start_time,'dd/MMM/yyyy hh:mm:ss') || 'NA');
			} catch (e) {
				arr.push('NA');
			}

			try {
				arr.push(this.datePipe.transform(o.end_time, 'dd/MMM/yyyy hh:mm:ss') || 'NA');
			} catch (e) {
				arr.push('NA');
			}

			try {
				arr.push(o.start_addr?.replace(/,/g,' ') || 'NA');
			} catch (e) {
				arr.push('NA');
			}

			try {
				arr.push(o.start?.latitude || 'NA');
			} catch (e) {
				arr.push('NA');
			}

			try {
				arr.push(o.start?.longitude || 'NA');
			} catch (e) {
				arr.push('NA');
			}

			try {
				arr.push(o?.nearest_landmark?.name && o.nearest_landmark?.dist ? o.nearest_landmark.dist/1000 + " KM from " + o.nearest_landmark.name : "NA");
			} catch (e) {
				arr.push('NA');
			}

			try {
				arr.push(o.top_speed || '0');
			} catch (e) {
				arr.push('0');
			}

			try {
				arr.push((o.duration/3600).toFixed(2) || "0");
			} catch (e) {
				arr.push("0");
			}

			try {
				arr.push((o.distance/1000).toFixed(2) || "0");;
			} catch (e) {
				arr.push('0');
			}
			return arr;
		}));
  }

  drawRoute() {
    if (this.playBackData && this.playBackData.length > 0) {
      this.playBackData?.forEach((data: any, idx:number) => {
        if(idx == 0 && data?.start?.latitude && data?.start?.longitude) {
          this.marker.lat = data?.start?.latitude;
          this.marker.lng = data?.start?.longitude;
        }
        if ((data?.tel || data?.imei) && data.points && data.points.length) {
          data.points?.forEach((point: any) => {
            let dist=point?.cum_dist/1000;
            point.lat = point.lat || point.latitude;
            point.lng = point.lng || point.longitude;
            if(point.lat && point.lng) {
              this.polyLine.latLng.push({ lat: point.lat, lng: point.lng, vehSpeed:point?.speed,course: point?.course,cum_dist:dist,datetime:point.datetime});
            }
          });
          if(this.selectedTripData?.tripBase === 'GPS Based') {
            this.preProcessing();
          }
        }
      });
    }
  }
  
  drawAlerts () {
    if(this.showAlerts) {
      let req: any = {};
      req.imei = [];
      if(this.code){
        req.code = this.code;
        if(req.code.includes('refill')){
          req.code.push('drain');
        }
      }
      req.imei.push(Number(this.device?.imei || this.selectedTripData?.device?.imei || this.selectedTripData?.vehicle?.device_imei || this.selectedTripData?.imei));
      req.from = this.selectedTripData?.start_date;
      req.to = this.selectedTripData?.end_date || new Date().toISOString();
      this.masterService.getAllAlerts(req).subscribe(alerts => {
        if(this.alertsData.length > 0) {

  secondsToHms(d : any) {
    d = Number(d);
    var h = Math.floor(d / 3600);
    var m = Math.floor(d % 3600 / 60);
    var s = Math.floor(d % 3600 % 60);

    var hDisplay = h > 0 ? h + (h == 1 ? " hour, " : " hours, ") : "";
    var mDisplay = m > 0 ? m + (m == 1 ? " minute, " : " minutes, ") : "";
    var sDisplay = s > 0 ? s + (s == 1 ? " second" : " seconds") : "";
    return hDisplay + mDisplay + sDisplay;
}


  drawFlags() {
    if(this.playBackData && this.playBackData.length > 0) {
      this.markers = [];

      this.playBackData?.forEach((data: any,idx:number) => {
        if(idx === 0) {
          this.markers.push({ lat: data?.start?.latitude, lng: data?.start?.longitude, icon:this.start});
        } else if(idx === length-1) {
          this.markers.push({ lat: data?.stop?.latitude, lng: data?.stop?.longitude, icon:this.stop});
        }
        let duration = this.secondsToHms(data?.duration);
        let idle=this.secondsToHms(data?.idle_duration);
        if(data.status === 'idle' && data.drive === false) {
          if(data?.stop?.latitude && data?.stop?.longitude) {
            this.markers.push({ lat: data?.stop?.latitude, lng: data?.stop?.longitude, icon:this.idleFlag, start_time: data?.start_time, end_time: data?.end_time,duration: duration, address:data?.start_addr || data?.stop_addr,landmark:data?.NearLandMark , idleDuration:idle});
          }
        } else if(data.drive === false || data.status === 'stopped') {
          if(data?.stop?.latitude && data?.stop?.longitude) {
            this.markers.push({ lat: data?.stop?.latitude, lng: data?.stop?.longitude, icon:this.stopFlag, start_time: data?.start_time, end_time: data?.end_time,duration: duration, address:data?.start_addr || data?.stop_addr,landmark:data?.NearLandMark , idleDuration:idle});

      });
    }

  speedChange() {
    // clear interval and statrting again so that
    // previous speed interval doesn't affect changed speed's delay
    if(this.timerRef) {
      clearInterval(this.timerRef);
      this.timerRef = undefined;
    }
    if (this.speed === 10) {
      this.delay = 100 * 10;
    } else if (this.speed === 20) {
      this.delay = 200 * 4;
    } else if (this.speed === 30) {
      this.delay = 300 * 2;
    } else if (this.speed === 40) {
      this.delay = 400 * 1;
    } else if (this.speed === 50) {
      this.delay = 500 / 5;
    } else if (this.speed === 60) {
      this.delay = 600 / 10;
    } else if (this.speed === 70) {
      this.delay = 700 / 15;
    } else if (this.speed === 80) {
      this.delay = 800 / 20;
    } else if (this.speed === 90) {
      this.delay = 900 / 25;
    } else if (this.speed === 100) {
      this.delay = 1000 / 30;

  }

  playRoute() {
    if(this.timerRef) return;
    // this.showRoute = false;
    this.showPause = true;
    if (true || !this.isPaused) {
      this.showTruck = true;
      this.idx = this.idx || 0;
      let len = this.markerLatLng.length;
      const mContext = this;
      mContext.timerRef = setInterval(() => {
        if(this.idx < len) {
            if(mContext.markerLatLng[this.idx].lat && mContext.markerLatLng[this.idx].lng) {
              mContext.marker.lat = mContext.markerLatLng[this.idx].lat;
              mContext.marker.lng = mContext.markerLatLng[this.idx].lng;
              mContext.reCenter.lat=mContext.marker.lat;
              mContext.reCenter.lng=mContext.marker.lng;
              mContext.reCenter.zoom=12;
            }
            mContext.marker.vehSpeed=mContext.polyLine.latLng[this.idx]?.vehSpeed || mContext.marker.vehSpeed;
            if(mContext.polyLine.latLng[this.idx]?.course) {
              mContext.setTruckIcon(mContext.polyLine.latLng[this.idx].course);
            }
            mContext.marker.distance=mContext.polyLine.latLng[this.idx]?.cum_dist;
            mContext.marker.dateTime=mContext.polyLine.latLng[this.idx]?.datetime;
            if(this.idx+5 < len) {
              this.idx += 5;
            } else {
              this.idx = len-1;
            }           
        } else {
          mContext.pauseRoute();
        }
        // console.info(mContext.delay);
      },mContext.delay);
    }
    this.isPaused = false;
  }

  preProcessing() {
    this.markerLatLng = [];
    if(this.polyLine.latLng.length < 2)  return;
    let len = this.polyLine.latLng.length;
    for(let i=1;i<len;i++) {
      let point1 : LatLng = {lat:this.polyLine.latLng[i-1].lat,lng:this.polyLine.latLng[i-1].lng};
      let point2 : LatLng = {lat:this.polyLine.latLng[i].lat,lng:this.polyLine.latLng[i].lng};
      let dist1_2 = this.getDistanceInKm(point1.lat,point1.lng,point2.lat,point2.lng);
      if(dist1_2 > 10) {
        console.info(dist1_2);
        this.deltaLat = (this.polyLine.latLng[i].lat - this.polyLine.latLng[i-1].lat)/this.numDeltas;
        this.deltaLng = (this.polyLine.latLng[i].lng - this.polyLine.latLng[i-1].lng)/this.numDeltas;
        for(let it=1;it<=this.numDeltas;it++){
          let newLat = this.polyLine.latLng[i-1]?.lat + it * this.deltaLat;
          let newLng = this.polyLine.latLng[i-1]?.lng + it * this.deltaLng;
          if(newLat && newLng) {
            this.markerLatLng.push({lat:newLat,lng: newLng});
          }
        }
      } else {
        this.markerLatLng.push(point2);
      }
    }
  }

  getDistanceInKm (lat1:number, lng1:number, lat2:number, lng2:number) : number {
		let R = 6371; // Radius of the earth in km
		let dLat = this.deg2rad(lat2 - lat1);  // deg2rad below
		let dLon = this.deg2rad(lng2 - lng1);
		let a =
			Math.sin(dLat / 2) * Math.sin(dLat / 2) +
			Math.cos(this.deg2rad(lat1)) * Math.cos(this.deg2rad(lat2)) *
			Math.sin(dLon / 2) * Math.sin(dLon / 2)
		;
		let c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a));
		let d = R * c; // Distance in km
		return d;
	};

	deg2rad(deg: number) : number{
		return deg * (Math.PI / 180)
	}
}
