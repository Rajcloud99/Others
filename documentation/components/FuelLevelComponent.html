import { Component, HostListener, OnInit, ViewEncapsulation } from '@angular/core';
import { ChartService } from 'src/app/services/chart.service';
import { MasterService } from '../master.service';
import { DatePipe } from '@angular/common';
import { differenceInCalendarDays } from 'date-fns';
import { MoreMenu } from 'src/app/shared/modal/moreMenu.modal';
import { CommonService } from 'src/app/services/common.service';

declare let d3: any;// it is used in html(nvd3) internally so don't remove.
@Component({
  selector: 'app-fuel-level',
  templateUrl: './fuel-level.component.html',
  styleUrls: ['./fuel-level.component.scss','../../../../node_modules/nvd3/build/nv.d3.css'],
  encapsulation: ViewEncapsulation.None

export class FuelLevelComponent implements OnInit {
  
  innerWidth = window.innerWidth;
  isMobFilterVisible = false;
  selectedVehicle: any;
  startDate: Date | null = null;
  endDate: Date | null = null;
  vehDataCopy: any = [];
  vehData: any = [];
  fuelLevelData: any = []
  f_options: any;
  lineChart: any;
  timeGap: any;// in minutes
  aTimeGap: any = [];
  fuelLevelDataCopy: any =  [];
  isListView = true;
  xLabelData: any = [];

  @HostListener("window: resize", ["$event"])
  onResize(event: any) {
    this.innerWidth = window.innerWidth;
  }
  
  constructor(
    private chartService: ChartService,
    private datePipe: DatePipe,
    private commonService: CommonService,
    private masterService: MasterService) {
    // for timewise
    this.f_options = this.chartService.getOptions({type:'lineChart',xAxisLabel:'Date and time',yAxisLabel:'Fuel Level (litre)',xData:this.xLabelData});
   }

  ngOnInit(): void {
    this.innerWidth = window.innerWidth;
    this.isListView = true;
    let _moreMenu: MoreMenu[] = [
      {
        tittle: "filter",
        iconName: "fa-filter",
        isVisible: true,
        clickEventMethode:()=> {
          this.showFilter();
        },
      },
      {
        tittle: 'Fuel Data',
        iconName: this.isListView ? "fa-th-list" : "fa-bar-chart",
        isVisible:true,
        clickEventMethode:()=>{
          this.gotoListView();
        }
      },
      {
        tittle:"Download",
        iconName:"fa-download",
        isVisible:true,
        clickEventMethode:()=>{
          this.downloadReport();
        }
      }  
    ];
    this.aTimeGap = [
      {
        name:"1 Min",
        value:1
      },
      {
        name:"5 Min",
        value:5
      },
      {
        name:"10 Min",
        value:10
      },
      {
        name:"15 Min",
        value:15
      },
      {
        name:"20 Min",
        value:20
      },
      {
        name:"25 Min",
        value:25
      },
      {
        name:"30 Min",
        value:30
      },
      {
        name:"1 hr",
        value:60
      }
    ];
    this.timeGap = this.aTimeGap[2];
    this.commonService.setNavTittleAndMenu("Fuel Level Data",_moreMenu);
    this.getVehicles();
  }

  showFilter() {
    this.isMobFilterVisible = true;
  }

  transformDate(date: any) {
    return this.datePipe.transform(date, 'd/M/yy, h:mm a');
  }

  gotoListView() {
    this.isListView = !this.isListView;
  }

  reset() {
    this.startDate = null;
    this.endDate = null;
    this.selectedVehicle = null;
  }

  downloadReport() {
    this.search(true);
  }
  disabledEndDate = (endValue: Date): boolean => {
    if (!endValue || !this.startDate) {
      return false;
    }
    let startDate = new Date(this.startDate);
    startDate.setDate(startDate.getDate()-1);// for 1 days selection
    // for disabling future date selection
    if(new Date().getTime() < endValue.getTime()) {
      endValue = new Date();
    }
    return endValue.getTime() < startDate.getTime() || differenceInCalendarDays(endValue, startDate) > 7;
  };

  disabledStartDate = (startValue: Date): boolean => {
    if (!startValue || !this.endDate) {
      return false;
    }
    let endDate = new Date(this.endDate);
    return startValue.getTime() > endDate.getTime() || differenceInCalendarDays(endDate, startValue) > 6;
  };

  searchVehicle(e:any) {
    let exp = String(e);
    this.vehDataCopy = this.vehData.filter((veh: any) => {return veh?.vehicle_reg_no?.toLowerCase().includes(exp.toLowerCase())});
  }
  
  search(value ?: any) {
    let req: any = {};
    req.from = this.startDate;
    req.to = this.endDate;
    if(value) req.download = value;
    req.device_id = [];
    req.device_imei = this.selectedVehicle?.device_imei;
    // req.device_imei = 357073295741991;
    req.vehicle_no = this.selectedVehicle?.vehicle_reg_no;
    req.timeGap = this.timeGap.value || 10;
    req.device_id.push(req.device_imei);
    delete req.device_imei;
    this.masterService.getAllFuelLevel(req).subscribe((data: any) => {
      if(data) {
        if(typeof data === 'string') {
          let url = data;
          this.commonService.downloadFileWithUrl(url);
          return;
        }
        if(data.length=== 0) {
          return this.commonService.error('No Data Found For Vehicle no. ' + this.selectedVehicle?.vehicle_reg_no);
        }
        this.fuelLevelData = data;
        // creating deep copy so that chnages in original array does'nt affect copied array
        this.fuelLevelDataCopy = JSON.parse(JSON.stringify(this.fuelLevelData));
        // sort fuel level data datetime wise ascending order
        this.fuelLevelData = this.fuelLevelData.sort(
          (a: any, b: any) => new Date(a.datetime).getTime() - new Date(b.datetime).getTime()
          );
        let i = 0;
        for(let fLvl of this.fuelLevelData) {
          fLvl.x = ++i;
          const date = this.transformDate(fLvl.datetime);
          this.xLabelData.push(date); 
          delete fLvl.datetime;
          fLvl.y = parseFloat(fLvl.f_lvl.toFixed(2)) || 0;
          delete fLvl.f_lvl;
          delete fLvl.fl;
        }
        this.lineChart = [];
        this.lineChart.push({key: 'Fuel Level', values: this.fuelLevelData});
      }
    });
  }

  getVehicles() {
    let req: any = {
      all: 'true',
			device_imei: {
				$exists: true,
				$ne: null
			},
			project: {
				device_imei: 1,
				vehicle_reg_no: 1,
				status: 1,
				segment_type: 1
			}
    };
    this.masterService.getAllVehicle(req).subscribe((data: any)=>{
      if(data){
        this.vehData = data;
        this.vehDataCopy = data;
      }
    })
  }
}
