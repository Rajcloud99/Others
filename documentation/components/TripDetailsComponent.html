  constructor(
    private route: ActivatedRoute,
    private masterService: MasterService,

    this.route.params.subscribe(queryParam => {
      if(queryParam._id) {
        const currentState = this.router.getCurrentNavigation();
        this.receivedData = [currentState && currentState.extras.state];
        this.selectedTrip = this.receivedData[0];
        const pipe = new FilterstringgetdatePipe();
        const mContext = this;
        this.allocationDate = pipe.transform(mContext.selectedTrip?.statuses,'Trip not started');
        this.startDate = pipe.transform(mContext.selectedTrip?.statuses,'Trip started');
        this.endDate = pipe.transform(mContext.selectedTrip?.statuses,'Trip ended');
      }
    })
  }

  clean(obj: any) {
    for (let propName in obj) {
      if (
        obj[propName] === null ||
        obj[propName] === undefined ||
        obj[propName] === ''
      ) {
        delete obj[propName];
      }
    }
  }

  disabledEndDate = (endValue: Date): boolean => {
    if (!endValue || !this.selectedTrip.start_date) {
      return false;
    }
    let startDate = new Date(this.selectedTrip.start_date);
    startDate.setDate(startDate.getDate() - 1);
    let today = new Date();
    return endValue.getTime() < startDate.getTime() || endValue.getTime() > today.getTime();
  };

  getCnstHist() {
    if(this.selectedTrip) {
      let reqBody: any = {};
      if(this.selectedTrip?.driver?.prim_contact_no)
        reqBody.mobile = this.selectedTrip.driver.prim_contact_no;
      this.masterService.getConsentHistory(reqBody).subscribe(data => {
        if(data) {
          this.cnstHistData = data;
        } else {
          this.cnstVisible = false;
        }
      })
    }
  }

  gotoLiveTrack() {
    this.router.navigateByUrl('home/tracking/mapView/liveTrack/'+this.selectedTrip?.device?.imei, {state : this.selectedTrip});
  }

    this.clipboard.copy(textToCopy);
    this.commonService.success('Link copied successfully');
  }

  genShareLink(template: TemplateRef<{}>) {
    let req: any = {};
    req.trip = this.selectedTrip?._id;
    req.expiry = Number(this.duration);
    this.masterService.genLinkTrip(req).subscribe(res => {
      if(res) {
        let oldBaseUrl = res.match(/^https?:\/\/[^#?\/]+/)[0];
        let newBaseUrl = window.location.origin;
        if(this.innerWidth<=500 || oldBaseUrl === newBaseUrl) {
          this.shareableLink = res;
        } else  {
          this.shareableLink = res.replace(oldBaseUrl,newBaseUrl);
        }
        this.sharePopUpVisible = false;
        this.notification.template(template,{nzDuration: 1000000000});
      }
    });
  }

  getDocStatus(isV2: any) {
    if (this.selectedTrip) {
      let reqBody = {
        _id: this.selectedTrip && this.selectedTrip._id,
        modelName: 'trip',
      };
      this.masterService.getDocuments(reqBody).subscribe((data: any) => {
        if (data) {
          this.receivedDocument = data.data.files;
          this.selectedTrip.receivedDocument = this.receivedDocument;
          if(isV2)
          this.openUpload(this.selectedTrip);
          else
            this.openUploadImagesPopUp(this.receivedDocument);
          // this.categoryMerge(this.receivedDocument);
        }else{
          if(isV2)
            this.openUpload(this.selectedTrip);
          else
            this.openUploadImagesPopUp(this.receivedDocument);
        }
      });
    }
  }

  updateTripInfo  (): void {
    let oSend : any = {};
    oSend.date  = this.date;
    oSend.date.setHours(this.time.getHours());
    oSend.date.setMinutes(this.time.getMinutes());
    oSend.date.setMilliseconds(0);
    if(new Date(this.date).getTime() < new Date(this.allocationDate).getTime()) {
      return this.commonService.errorModal('Error','Start date can not be before the allocation date');
    }
    if(new Date(this.date).getTime() < new Date(this.startDate).getTime()) {
      return this.commonService.errorModal('Error','Start date can not be after than end date');
    }
    oSend.status = this.selectedTrip.updatedStatus || this.aStatusChange[this.aStatusChange.indexOf(this.selectedTrip.status)+1];
    oSend.remark = this.tripRemark || this.selectedTrip.remark;
    oSend.updated_status = this.selectedTrip.updated_status || {};

    if (this.selectedTrip.updatedStatus === 'Trip started') {
      if (this.selectedTrip.route) {
        oSend.route_name = this.selectedTrip.route.name;
        oSend.route = this.selectedTrip.route._id;
      }
      oSend.trip = {};
      oSend.trip.gr = this.selectedTrip.gr.map((curr:any) => ({
        _id: curr._id,
        consignee: curr.consignee,
        consignor: curr.consignor,
        status: 'GR Assigned'
      }));
    }
    this.isDisable = true;
    this.masterService.updateTripStatus(this.selectedTrip._id,oSend).subscribe((data: any) => {
      if(data)
        this.router.navigateByUrl('home/operation/tripDetails');
      this.visible = false;
    });
  }

  activityReport() {
    let request : any = {};
    request.download=true;
    let deviceIMEI:any = [];
    let regVeh = [];
    if(!(this.selectedTrip && this.selectedTrip.device && this.selectedTrip.device.imei)){
      return this.commonService.error('IMEI not available on this trip');
    }
    if(!(this.selectedTrip && this.selectedTrip.start_date)){
      return  this.commonService.error("trip start date not available on this trip");
    }
    deviceIMEI.push(this.selectedTrip.device.imei);
    regVeh.push({reg_no : this.selectedTrip.vehicle_no ,device_imei:this.selectedTrip.device.imei});
    request.device_id = deviceIMEI;
    request.regVeh = regVeh;
    request.start_time = this.selectedTrip.start_date;
    request.end_time = this.selectedTrip.end_time ||   new Date();
    request.reportType='report_activity';
    let apiType='activity';
      this.masterService.getGpsReports(request,apiType).subscribe((data: any)=>{
        if(data){
          let url = data.data || data.url;
          if(Object.keys(url).length>0 && typeof url === 'string'){
            this.commonService.downloadFileWithUrl(url);
        }
      }
      })

}
