  secondsToHms(d : any) {
    d = Number(d);
    var h = Math.floor(d / 3600);
    var m = Math.floor(d % 3600 / 60);
    var s = Math.floor(d % 3600 % 60);

    var hDisplay = h > 0 ? h + (h == 1 ? " hour, " : " hours, ") : "";
    var mDisplay = m > 0 ? m + (m == 1 ? " minute, " : " minutes, ") : "";
    var sDisplay = s > 0 ? s + (s == 1 ? " second" : " seconds") : "";
    return hDisplay + mDisplay + sDisplay;
}


  drawFlags() {
    if(this.playBackData && this.playBackData.length > 0) {
      this.markers = [];

      this.playBackData?.forEach((data: any,idx:number) => {
        if(idx === 0) {
          this.markers.push({ lat: data?.start?.latitude, lng: data?.start?.longitude, icon:this.start});

          this.markers.push({ lat: data?.stop?.latitude, lng: data?.stop?.longitude, icon:this.stop});
        }
        let duration = this.secondsToHms(data?.duration);
        let idle=this.secondsToHms(data?.idle_duration);
        if(data.status === 'idle' && data.drive === false) {
          if(data?.stop?.latitude && data?.stop?.longitude) {
            this.markers.push({ lat: data?.stop?.latitude, lng: data?.stop?.longitude, icon:this.idleFlag, start_time: data?.start_time, end_time: data?.end_time,duration: duration, address:data?.start_addr || data?.stop_addr,landmark:data?.NearLandMark , idleDuration:idle});
          }
        } else if(data.drive === false || data.status === 'stopped') {
          if(data?.stop?.latitude && data?.stop?.longitude) {
            this.markers.push({ lat: data?.stop?.latitude, lng: data?.stop?.longitude, icon:this.stopFlag, start_time: data?.start_time, end_time: data?.end_time,duration: duration, address:data?.start_addr || data?.stop_addr,landmark:data?.NearLandMark , idleDuration:idle});
          }
        }
      });
    }

  speedChange() {
    // clear interval and statrting again so that
    // previous speed interval doesn't affect changed speed's delay

      clearInterval(this.timerRef);

    }
    if (this.speed === 10) {
      this.delay = 100 * 10;
    } else if (this.speed === 20) {
      this.delay = 200 * 4;

    } else if (this.speed === 40) {
      this.delay = 400 * 1;
    } else if (this.speed === 50) {
      this.delay = 500 / 5;
    } else if (this.speed === 60) {
      this.delay = 600 / 10;
    } else if (this.speed === 70) {

  playRoute() {
    if(this.timerRef) return;
    // this.showRoute = false;
    this.showPause = true;
    if (true || !this.isPaused) {
      this.showTruck = true;
      this.idx = this.idx || 0;
      let len = this.markerLatLng.length;
      const mContext = this;
      mContext.timerRef = setInterval(() => {
        if(this.idx < len) {
            if(mContext.markerLatLng[this.idx].lat && mContext.markerLatLng[this.idx].lng) {
              mContext.marker.lat = mContext.markerLatLng[this.idx].lat;
              mContext.marker.lng = mContext.markerLatLng[this.idx].lng;
              mContext.reCenter.lat=mContext.marker.lat;
              mContext.reCenter.lng=mContext.marker.lng;
              mContext.reCenter.zoom=12;
            }
            mContext.marker.vehSpeed=mContext.polyLine.latLng[this.idx]?.vehSpeed || mContext.marker.vehSpeed;
            if(mContext.polyLine.latLng[this.idx]?.course) {
              mContext.setTruckIcon(mContext.polyLine.latLng[this.idx].course);
            }
            mContext.marker.distance=mContext.polyLine.latLng[this.idx]?.cum_dist;
            mContext.marker.dateTime=mContext.polyLine.latLng[this.idx]?.datetime;
            if(this.idx+5 < len) {
              this.idx += 5;
            } else {
              this.idx = len-1;
            }           
        } else {
          mContext.pauseRoute();
        }
        // console.info(mContext.delay);
      },mContext.delay);
    }
    this.isPaused = false;
  }

  // to add more points in polyline if points are found at more than 10km distance
  // and creates marker latLng array to traverse by vehicle
  preProcessing() {
    this.markerLatLng = [];
    if(this.polyLine.latLng.length < 2)  return;
    let len = this.polyLine.latLng.length;
    for(let i=1;i<len;i++) {
      let point1 : LatLng = {lat:this.polyLine.latLng[i-1].lat,lng:this.polyLine.latLng[i-1].lng};
      let point2 : LatLng = {lat:this.polyLine.latLng[i].lat,lng:this.polyLine.latLng[i].lng};
      let dist1_2 = this.getDistanceInKm(point1.lat,point1.lng,point2.lat,point2.lng);
      if(dist1_2 > 10) {
        console.info(dist1_2);
        this.deltaLat = (this.polyLine.latLng[i].lat - this.polyLine.latLng[i-1].lat)/this.numDeltas;
        this.deltaLng = (this.polyLine.latLng[i].lng - this.polyLine.latLng[i-1].lng)/this.numDeltas;
        for(let it=1;it<=this.numDeltas;it++){
          let newLat = this.polyLine.latLng[i-1]?.lat + it * this.deltaLat;
          let newLng = this.polyLine.latLng[i-1]?.lng + it * this.deltaLng;
          if(newLat && newLng) {
            this.markerLatLng.push({lat:newLat,lng: newLng});
          }
        }
      } else {
        this.markerLatLng.push(point2);
      }
    }
  }

  getDistanceInKm (lat1:number, lng1:number, lat2:number, lng2:number) : number {
		let R = 6371; // Radius of the earth in km
		let dLat = this.deg2rad(lat2 - lat1);  // deg2rad below
		let dLon = this.deg2rad(lng2 - lng1);
		let a =
			Math.sin(dLat / 2) * Math.sin(dLat / 2) +
			Math.cos(this.deg2rad(lat1)) * Math.cos(this.deg2rad(lat2)) *
			Math.sin(dLon / 2) * Math.sin(dLon / 2)
		;
		let c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a));
		let d = R * c; // Distance in km
		return d;
	};

	deg2rad(deg: number) : number{
		return deg * (Math.PI / 180)
	}
