import { Component, OnInit } from '@angular/core';
import { ActivatedRoute } from '@angular/router';
import { MasterService } from 'src/app/modules/master.service';
import { ConstantService } from 'src/app/services/constant.service';
import { StorageService } from 'src/app/services/storage.service';

interface Marker {
  lat: number;
  lng: number;
  icon?: string;
  label?: string;
  eta?: string;
  address?: string;
  draggable?: boolean;
  etaUnit?: string;
  dist?: number|string;
  distUnit?: string;
}

@Component({
  selector: 'app-share-location',
  templateUrl: './share-location.component.html',
  styleUrls: ['./share-location.component.scss']
})
export class ShareLocationComponent implements OnInit {
  vehicleId: any;
  vehicleData: any;
  vehStorage:any;
  truck_path: string;
  truckIcon: any;
  isOpen = true;
  selectedVehicle: any;

  constructor(
    private masterService: MasterService,
    private constantService: ConstantService,
    private storageService: StorageService,
    private route: ActivatedRoute
  ) {
    this.truck_path = this.constantService.constants.truck_path;
   }
  marker: Marker = {
    lat: 23.673858,
    lng: 83.815982,
  };
  zoom: number = 8;
  ngOnInit(): void {
    this.truckIcon = {
      path: this.truck_path,
      fillColor: "#000000",
      fillOpacity: 0.8,
      strokeWeight: 0.5,
      scale: 0.5,
      rotation: 90
    };
    this.vehStorage = this.storageService.get('allVehicles');
    this.route.queryParams.subscribe(params => {
      this.vehicleId = params['vehicle'];
      console.info('vehicle id '+this.vehicleId);
    });
    this.getVehicleData();
  }

  changeTruckIcon(data:any){
    let vehId = data && data?.length && data[0]?._id;
    let foundVeh=this.vehStorage && this.vehStorage.length && this.vehStorage.filter((d:any)=>{
      return vehId==d?._id;
      })
    if(!(foundVeh && foundVeh.length)){
      return;
    }  
    let status = this.vehicleData[0]?.gpsData?.status;
    this.truckIcon = {
      path: this.truck_path,
      fillColor: status=='stopped'?"#FF0000": status=='running'? "#15e425":"#939182",
      fillOpacity: 0.8,
      strokeWeight: 0.5,
      scale: 0.5,
      rotation: this.vehicleData[0]?.gpsData?.course
    };
  }

  setStatus() {    
   for(let veh of this.vehicleData) {
     let positionTime : number = new Date(veh.gpsData?.positioning_time).getTime();
     let locationTime: number = new Date(veh.gpsData?.location_time).getTime();
     let stoppageTime = 5;//its data sync frequency
     let ptDiffMin = Math.ceil((new Date().getTime() - positionTime) / 60000); // in Min
     let ltDiffMin = Math.ceil((new Date().getTime() - locationTime) / 60000); // in Min
     if (ptDiffMin && ptDiffMin < 300) {//5 hr no offline
       if (ltDiffMin && ltDiffMin <= stoppageTime && veh.gpsData?.speed > 0) {
        veh.gpsData.status = "running";
       } else {
         veh.gpsData.status = "stopped";
       }
     } else {
       veh.gpsData = veh.gpsData || {};
       veh.gpsData.status = "offline";
     }
   }
  }

  getVehicleData() {
    let req: any = {};
    req.vehicle = this.vehicleId;
    req.byPass = true;
    this.masterService.getPublicLocationData(req).subscribe(res => {
      if(res) {
        this.vehicleData = res;
        this.setStatus();
        this.changeTruckIcon(this.vehicleData);
        this.selectedVehicle = res[0];
        this.marker.lat = this.selectedVehicle.gpsData.lat;
        this.marker.lng = this.selectedVehicle.gpsData.lng;
      }
    });
  }

}
