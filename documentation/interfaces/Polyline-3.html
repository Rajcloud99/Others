import { Component, OnInit} from '@angular/core';
import { ActivatedRoute, Router } from '@angular/router';
import { NzMarks } from 'ng-zorro-antd/slider';
import { CommonService } from 'src/app/services/common.service';

interface Marker {
  lat: number;
  lng: number;
  icon?: string;
  label?: string;
  eta?: string;
  address?: string;
  draggable?: boolean;
  etaUnit?: string;
  dist?: number | string;
  distUnit?: string;
  vehSpeed?:number;
  course?:number;
  distance?:number;
  dateTime?:Date;
}

  lat: number;

@Component({
  selector: 'app-milestone-view',
  templateUrl: './milestone-view.component.html',
  styleUrls: ['./milestone-view.component.scss']
})
export class MilestoneViewComponent implements OnInit{
  innerWidth = window.innerWidth;
  marks: NzMarks = {
    10: '10',
    50: '50',
    100: '100',
  };
  map: any;
  isOpen:boolean=false;
  zoom: number = 7;
  marker: Marker = {
    lat: 23.673858,
    lng: 83.815982,
    vehSpeed:0,
  };
  lat: any = 23.673858;
  lng: any = 83.815982;
  routedata:any;
  start = 'assets/start.png';
  stop = 'assets/stop.png';
  stopFlag = 'assets/stopFlag.png';
  iconUrl: any;
  MilestoneMarker :any;
  SourceDestMarker :any;
  constructor(
    private route: ActivatedRoute,
    private router: Router,
    private commonService:  CommonService,
  ) {
    this.route.params.subscribe((queryParam) => {
      if (queryParam) {
        const currentState = this.router.getCurrentNavigation();
        this.routedata = currentState && currentState.extras.state;
      }
    });
  }

  ngOnInit(): void {
    this.commonService.setNavTittleAndMenu('Milestone View');
    if(this.routedata){
      if(this.routedata && this.routedata.milestone && this.routedata.milestone.length>0){
        this.MilestoneMarker = this.routedata.milestone;
      }
      this.SourceDestMarker=[];
      if(this.routedata && this.routedata.source){
        this.SourceDestMarker.push(this.routedata.source);
      }
      if(this.routedata && this.routedata.destination){
        this.SourceDestMarker.push(this.routedata.destination);
      }
    }
  }

}
