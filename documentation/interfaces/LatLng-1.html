export let browserRefresh = false;
@Component({
  selector: 'app-map',
  templateUrl: './map.component.html',
  styleUrls: ['./map.component.scss']
})
export class MapComponent implements OnChanges ,OnInit, OnDestroy {

  @Input() mapData: any;// to get data from any other parent component
  zoom: number = 8;
  isOpen = false;
  markers: any [] = [];
  visible = false;
  receivedData: any;
  vehData: any;
  roles: any;
  subscription: Subscription;
  innerWidth = window.innerWidth;
  selectedVehicle_vehicle_no: any;
  oldVehData: any = [];
  truck_path: any;
  truckIcon: any;
  marker: Marker = {
    lat: 23.673858,
    lng: 83.815982,
  };
  setColor = {"running": "#15e425", "online": "#15e425", "stopped": "red", "offline": "grey"};
  filteredMarker: any;
  sharePopUpVisible = false;
  shareableLink: any;
  ngOnChanges(changes: SimpleChanges) {
    if(changes.mapData.currentValue) {
      this.marker.lat = changes.mapData.currentValue.vehicle.lat;
      this.marker.lng = changes.mapData.currentValue.vehicle.lng;
      this.selectedVehicle_vehicle_no = changes.mapData.currentValue.vehicle_reg_no;
      this.zoom = 14;
      this.closeAllMarkers();

      // console.log(changes.mapData);
      // console.log(this.markers);
      this.filteredMarker = this.markers.find((item: any)=>{return item.vehicle_reg_no === this.selectedVehicle_vehicle_no});
      this.filteredMarker.isOpen = true;
    }
  }
  constructor(
    private masterService: MasterService,
    private storageService: StorageService,
    private router: Router,
    private socket: Socket,
    private modal: NzModalService,
    private commonService: CommonService,
    private viewContainerRef: ViewContainerRef,
    private constantService: ConstantService,
    private notification: NzNotificationService,
    private clipboard: Clipboard,
  ) {
    // set empty map zoom level
    if(this.markers.length === 0) {
      this.zoom = 5;
    }
    const currentState = this.router.getCurrentNavigation();
    this.vehData = currentState && currentState?.extras?.state;
    this.truck_path = this.constantService.constants.truck_path;
    if(!this.vehData) {
      this.getVehiclewiseData();
    } else if(this.vehData) {
      this.oldVehData = this.vehData;
      this.vehData = [];
      if(this.oldVehData) {
        for (const [keys, value] of Object.entries(this.oldVehData)) {
          this.vehData.push(value);
        }
      }
      if(this.vehData.length > 0) {
        this.vehData.slice(-1);
      }
      this.setMarkers();
      if(this.markers.length > 0) {
        this.marker.lat = this.oldVehData[0].vehicle.gpsData.lat;
        this.marker.lng = this.oldVehData[0].vehicle.gpsData.lng;
        this.zoom = 14;
      } else {
        this.zoom = 4;
      }
      this.closeAllMarkers();

      // console.log(changes.mapData);
      this.filteredMarker = this.markers.find((item: any)=>{return item.lat == this.marker.lat && item.lng == this.marker.lng});
      if(this.filteredMarker) {
        this.filteredMarker.isOpen = true;
      }
    }
    this.subscription = router.events.subscribe((event) => {
      if (event instanceof NavigationStart) {
        browserRefresh = !router.navigated;
      }
  });
  }
  ngOnDestroy(): void {
    this.subscription.unsubscribe();
  }

  ngOnInit(): void {
    this.roles = this.storageService.get('userData').access;
  }

  closeAllMarkers() {
    for(const marker of this.markers) {
      marker.isOpen = false;
    }
  }

  copyText(textToCopy: string) {
    this.clipboard.copy(textToCopy);
    this.commonService.success('Link copied successfully');
  }

  genShareLink(selectedVehicle:any,template: TemplateRef<{}>) {
    let req: any = {};
    req.vehicle = selectedVehicle?._id;
    this.masterService.genLinkLocation(req).subscribe(res => {
      if(res) {
        let oldBaseUrl = res.match(/^https?:\/\/[^#?\/]+/)[0];
        let newBaseUrl = window.location.origin;
        if(this.innerWidth<=500 || oldBaseUrl === newBaseUrl) {
          this.shareableLink = res;
        } else {
          this.shareableLink = res.replace(oldBaseUrl,newBaseUrl);
        }
        this.sharePopUpVisible = false;
        this.notification.template(template,{nzDuration: 1000000000});
      }
    });
  }

  setStatus() {
   for(let veh of this.vehData) {
     let positionTime : number = new Date(veh.vehicle?.gpsData?.positioning_time).getTime();
     let locationTime: number = new Date(veh.vehicle?.gpsData?.location_time).getTime();
     let stoppageTime = 10;
     let ptDiffMin = Math.ceil((new Date().getTime() - positionTime) / 60000); // in Min
     let ltDiffMin = Math.ceil((new Date().getTime() - locationTime) / 60000); // in Min
     if (ptDiffMin && ptDiffMin < 300) {//5 hr no offline
       if (ltDiffMin && ltDiffMin <= stoppageTime && veh.vehicle?.gpsData?.speed > 0) {
        veh.vehicle.gpsData.status = "running";
       } else {
         veh.vehicle.gpsData.status = "stopped";
       }
     } else {
       veh.vehicle.gpsData = veh.vehicle.gpsData || {};
       veh.vehicle.gpsData.status = "offline";
     }
   }
  }

  setMarkers() {
    this.vehData?.forEach((item: any)=>{
      if(item?.vehicle?.device_imei && item.vehicle?.gpsData?.lat && item.vehicle?.gpsData?.lng) {
        if(item?.vehicle?.gpsData?.status === 'stopped'){
          let color = this.setColor['stopped'];
          this.setTruckIcon(item?.vehicle?.gpsData?.course,color);
        } else if(item?.vehicle?.gpsData?.status == 'running') {
          let color = this.setColor['running'];
          this.setTruckIcon(item?.vehicle?.gpsData?.course,color);
        } else if(item?.vehicle?.gpsData?.status == 'offline') {
          let color = this.setColor['offline'];
          this.setTruckIcon(item?.vehicle?.gpsData?.course,color);
        }
      }
      if(item.vehicle?.gpsData?.lat && item.vehicle?.gpsData?.lng) {
        this.markers.push({lat:item.vehicle?.gpsData?.lat,lng:item.vehicle?.gpsData?.lng,icon:this.truckIcon,isOpen: false,vehicle_reg_no:item.vehicle?.vehicle_reg_no});
      }
    });
  }

    setTruckIcon(angle: any,color: any) {
      this.truckIcon = {
        path: this.truck_path,
        fillColor: color || "#15e425",
        fillOpacity: 0.8,
        strokeWeight: 0.5,
        scale: 0.5,
        rotation: angle || 90
      };
    }

   getVehiclewiseData() {
     this.masterService.vehicleWise({}).subscribe((res: any)=>{
      if(res && res.data) {
        this.receivedData = res;
        this.vehData = this.receivedData?.data;
        this.storageService.put('allVehicles',this.vehData);
        // this.vehData.forEach((item: any)=>{
        //   if(item?.vehicle?.device_imei && item.vehicle?.gpsData?.lat && item.vehicle?.gpsData?.lng)
        //   this.markers.push({lat:item.vehicle?.gpsData?.lat,lng:item.vehicle?.gpsData?.lng});
        // });
        this.setStatus();
        this.setMarkers();
      }
     });
   }

   gotoLiveTrack(selectedVehicle: any) {
     this.router.navigateByUrl('home/tracking/mapView/liveTrack/'+selectedVehicle?.device_imei,{state: selectedVehicle});
   }

   socketResponse() {
    this.socket.on('connect',function(args: any) {
      console.log("successful connection",args);
    });
    this.socket.on("connect_error",function(args: any){
      console.log("connect_error",args);
    });
    this.socket.on("connect_timeout",function(args: any){
      console.log("connect_timeout",args);
    });
   }

   onClose() {
     this.visible = false;
   }
}
